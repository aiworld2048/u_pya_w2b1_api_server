    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const notificationServer = @json($notificationServer);
            const authUserId = @json(auth()->id());

            if (!notificationServer || typeof io === 'undefined') {
                return;
            }

            const depositListEl = document.getElementById('depositNotificationList');
            const withdrawListEl = document.getElementById('withdrawNotificationList');
            const depositCountEl = document.getElementById('depositNotificationCount');
            const withdrawCountEl = document.getElementById('withdrawNotificationCount');
            const maxItems = 6;

            const listMap = {
                deposit: { list: depositListEl, count: depositCountEl },
                withdraw: { list: withdrawListEl, count: withdrawCountEl },
            };

            const buildUrl = (path) => {
                if (!path || path === '#') {
                    return '#';
                }

                if (path.startsWith('http')) {
                    return path;
                }

                return window.location.origin + (path.startsWith('/') ? path : '/' + path);
            };

            const removeEmptyState = (list, type) => {
                if (!list) {
                    return;
                }

                const empty = list.querySelector(`[data-empty-state="${type}"]`);
                if (empty) {
                    empty.remove();
                }
            };

            const pushNotification = (type, payload) => {
                const map = listMap[type] || listMap.deposit;
                if (!map.list) {
                    return;
                }

                const data = payload?.notification_data || {};
                const meta = data.meta || {};
                const playerName = data.player_name || payload?.player_name || 'Player';
                const rawAmount = meta.amount ?? data.amount ?? payload?.amount ?? null;
                const amountLabel = rawAmount !== null
                    ? `${Number(rawAmount).toLocaleString()} Ks`
                    : null;
                const bodyText = payload?.body || data.body || data.message || meta.message ||
                    `${playerName} submitted a ${type} request.`;
                const route = buildUrl(data.route || meta.route || '#');

                removeEmptyState(map.list, type);

                const wrapper = document.createElement('div');
                wrapper.className = 'dropdown-item py-2 border-bottom';
                wrapper.setAttribute('data-notification-item', 'true');

                const anchor = document.createElement('a');
                anchor.href = route;
                anchor.className = 'text-decoration-none d-block text-reset';

                const header = document.createElement('div');
                header.className = 'd-flex justify-content-between align-items-center mb-1';

                const nameEl = document.createElement('strong');
                nameEl.className = 'text-dark';
                nameEl.textContent = playerName;
                header.appendChild(nameEl);

                if (amountLabel) {
                    const badge = document.createElement('span');
                    badge.className = 'badge bg-light text-dark';
                    badge.textContent = amountLabel;
                    header.appendChild(badge);
                }

                const messageEl = document.createElement('small');
                messageEl.className = 'd-block text-secondary';
                messageEl.textContent = bodyText;

                const timeEl = document.createElement('small');
                timeEl.className = 'text-muted';
                timeEl.textContent = new Date().toLocaleTimeString();

                anchor.appendChild(header);
                anchor.appendChild(messageEl);
                anchor.appendChild(timeEl);

                wrapper.appendChild(anchor);

                map.list.prepend(wrapper);

                const items = map.list.querySelectorAll('[data-notification-item]');
                if (items.length > maxItems) {
                    items[items.length - 1].remove();
                }

                if (map.count) {
                    const current = parseInt(map.count.textContent, 10) || 0;
                    map.count.textContent = current + 1;
                }

                if (window.toastr) {
                    const toastTitle = type === 'withdraw' ? 'Withdraw Request' : 'Deposit Request';
                    window.toastr.info(bodyText, toastTitle);
                }
            };

            const socket = io(notificationServer, {
                transports: ['websocket', 'polling'],
            });

            socket.on('connect', function() {
                if (authUserId) {
                    socket.emit('register', authUserId);
                }
            });

            socket.on('receive_noti', function(payload) {
                const type = payload?.notification_data?.type || payload?.type || 'deposit';
                const normalizedType = type === 'withdraw' ? 'withdraw' : 'deposit';
                pushNotification(normalizedType, payload);
            });

            socket.on('disconnect', function() {
                console.warn('Disconnected from notification server');
            });
        });
    </script>
